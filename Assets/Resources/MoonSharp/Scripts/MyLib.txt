function Talk(Chara, Words)
   g_TalkChara = Chara
   g_TalkWords = Words

   return {g_TalkChara, g_TalkWords}
end
------------------------------------------------
-- セリフ発言用コマンド（質問付き）
------------------------------------------------
function Ask(Chara, Words, ...)
   g_TalkChara = Chara
   g_TalkWords = Words
   KeyWait()
   
   for i = 1, select("#", ...) do
	  ask = select(i, ...)
	  obj.addAsk(ask)
   end
end
